# Read ABS file
abs_raw <- read_excel(
  "data/allgroupsbyhouseholds.xlsx",
  sheet = "Data1",
  col_names = FALSE
) |>
  clean_names()

# Data rows start at row 11
date_row_idx <- 11

# Extract Employee households all-groups index
abs_lci <- abs_raw |>
  slice(date_row_idx:n()) |>
  transmute(
    date_raw = as.numeric(x1),
    employee_lci = as.numeric(x3)
  ) |>
  mutate(
    date = as.Date(date_raw, origin = "1899-12-30"),
    quarter = paste(format(date, "%Y"), paste0("Q", quarter(date)))
  ) |>
  select(quarter, employee_lci)

# Keep 2024-2025 only
abs_lci_2425 <- abs_lci |>
  filter(quarter %in% c(
    "2024 Q1", "2024 Q2", "2024 Q3", "2024 Q4",
    "2025 Q1", "2025 Q2", "2025 Q3", "2025 Q4"
  ))

# Merge

occupational_summary <- itw_all |>
  filter(!is.na(a18), a18 %in% c(1, 2)) |>

f8_summary <- itw_all |>
  select(quarter, f8_1, f8_2, f8_3) |>
  pivot_longer(cols = starts_with("f8"), 
               names_to = "category", 
               values_to = "rating") |>
  filter(!is.na(rating)) |>
  mutate(
    category = case_when(
      category == "f8_1" ~ "Housing (Rent/Mortgage)",
      category == "f8_2" ~ "Utilities(Electricity/Gas/Water)",
      category == "f8_3" ~ "Food"
    ),
    # Convert numeric 1-5 to readable labels
    rating_label = factor(rating, 
                          levels = 1:5, 
                          labels = c("1 - Very Difficult", "2", "3", "4", "5 - Very Easy"))
  ) |>
  group_by(quarter, category, rating_label) |>
  summarise(count = n(), .groups = "drop") |>
  group_by(quarter, category) |>
  mutate(pct = (count / sum(count)) * 100)
ggplot(f8_summary, aes(x = quarter, y = pct, fill = rating_label)) +
  geom_bar(stat = "identity", position = "fill") +

    title = "Ease of Meeting Essential Living Costs",
