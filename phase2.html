# Data rows start at row 11
date_row_idx <- 11

# Extract Employee households all-groups index
abs_lci <- abs_raw |>
  slice(date_row_idx:n()) |>
  transmute(
    date_raw = as.numeric(x1),
    employee_lci = as.numeric(x3)
  ) |>
  mutate(
    date = as.Date(date_raw, origin = "1899-12-30"),
    quarter = paste(format(date, "%Y"), paste0("Q", quarter(date)))
  ) |>
  select(quarter, employee_lci)

# Keep 2024-2025 only
abs_lci_2425 <- abs_lci |>
  filter(quarter %in% c(
    "2024 Q1", "2024 Q2", "2024 Q3", "2024 Q4",
    "2025 Q1", "2025 Q2", "2025 Q3", "2025 Q4"
  ))

# Merge
phase2_dataset <- wave_summary |>
  left_join(abs_lci_2425, by = "quarter")
