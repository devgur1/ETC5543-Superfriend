score_sf <- function(df, items) {

  m <- rowMeans(df[, items], na.rm = TRUE)
  s <- (m - 1) * 25
  s[is.nan(m)] <- NA
  s
}

Aggregation helper

agg_qtr_year <- function(df, var){
  df |>
    group_by(Year, Quarter) |>
    summarise(
      mean = mean(.data[[var]], na.rm=TRUE),
      sd   = sd(.data[[var]], na.rm=TRUE),
      n    = sum(!is.na(.data[[var]])),
      se   = sd/sqrt(n),
      lci  = mean - 1.96*se,
      uci  = mean + 1.96*se,
      .groups="drop"
    )
}

run_lm_trend <- function(df,outcome){

  lm(as.formula(paste(outcome,"~ Quarter_num")),
     data=df) |>
    tidy() |>
    filter(term=="Quarter_num") |>
    mutate(outcome=outcome)

}

run_ttest <- function(df,outcome){

  df |>
    filter(Quarter %in% c("Q1","Q4")) |>
    t.test(as.formula(paste(outcome,"~ Quarter")),
           data=_) |>
    tidy() |>
    mutate(outcome=outcome)

}
